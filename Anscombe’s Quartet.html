Reproducibility & Code

The following Python script reproduces all analyses and figures.


# anscombe_eda.py
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import statsmodels.api as sm

df = pd.read_csv("anscombe.csv")

def summary_stats(g):
    n = g.shape[0]
    mean_x = g['x'].mean()
    mean_y = g['y'].mean()
    var_x = g['x'].var(ddof=1)
    var_y = g['y'].var(ddof=1)
    std_x = g['x'].std(ddof=1)
    std_y = g['y'].std(ddof=1)
    cov = g[['x','y']].cov().iloc[0,1]
    corr = g['x'].corr(g['y'])
    X = sm.add_constant(g['x'])
    model = sm.OLS(g['y'], X).fit()
    slope = model.params['x']
    intercept = model.params['const']
    r2 = model.rsquared
    return pd.Series({'mean_x':mean_x, 'mean_y':mean_y, 'r':corr, 'slope':slope, 'r2':r2})

summary = df.groupby('dataset').apply(summary_stats)
summary.to_csv("output/summary_table.csv")

sns.lmplot(data=df, x='x', y='y', col='dataset', col_wrap=2, height=4)
plt.savefig("output/anscombe_faceted.png")
